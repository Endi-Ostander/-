section .multiboot
align 8
header_start:
    dd 0xe85250d6           ; magic Multiboot2
    dd 0                    ; architecture (0 = i386)
    dd header_end - header_start ; length
    dd -(0xe85250d6 + 0 + (header_end - header_start)) ; checksum (32-bit wraparound)

    ; End tag
    dw 0    ; type
    dw 0    ; flags
    dd 8    ; size
header_end:

section .note.GNU-stack noalloc noexec nowrite align=1
